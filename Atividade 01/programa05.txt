ALGORITMO

    DEFINA
        numero, quadrado, cubo real

    ENQUANTO (Verdadeiro)

        ESCREVA("Digite um numero maior que zero: ")
        leia numero
        
        SE (numero<=0)
            ENTAO
                ESCREVA("Digite um numero maior que zero!!")
        SENAO
            QUEBRE

    quadrado <- numero*numero
    cubo <- numero*numero*numero

    ESCREVA("Seu número ao quadrado é: ", quadrado)
    ESCREVA("Seu número ao cubo é: ", cubo)

FIM_ALGORITMO
